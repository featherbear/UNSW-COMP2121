p = 0x33
count = 0
max = 0xffff4345
fin = False

  
def increase():
  global p, count, max, fin
  count += 1
  if count == max:
    fin = True


def run():
  global p, count, max, fin
      
  carry = (0b10000000 & p) >> 7
  p = 0xFF & (p << 1)
  
  while fin == False:
    if carry == 0:
      increase()
    else:
      p = 0xFF & (p + 1)
      increase()
